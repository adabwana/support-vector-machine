(def auto-cat
  (-> auto
      (tc/map-columns :mpg-cat [:mpg]
                      #(if (>= % (stats/median (:mpg auto))) 1 0))

3.5 Evaluate model

(defn eval-list [model]
  (let [pred (predict model test-data)
        obs (factor ($ test-data 'mpg.cat))
